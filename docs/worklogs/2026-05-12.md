   │   └── database.py          # SQLAlchemy async engine + Base + get_db
   ├── models/
   │   ├── __init__.py
   │   ├── document.py          # Document ORM (PK: doc_id String36)
   │   └── layout_block.py      # LayoutBlock ORM (FK→documents)
   ├── repository/
   │   ├── __init__.py
   │   ├── base.py              # BaseRepository[T] 泛型 CRUD
   │   ├── document_repo.py     # +update_status, get_by_path
   │   └── layout_repo.py       # +get_by_page, get_by_doc, delete_by_doc
   └── services/
       ├── __init__.py
       └── mineru_service.py    # MinerU 封装 (parse_pdf, extract_formulas)

   ├── __init__.py

   tests/ (空目录)
   data/{raw,parsed,exports}/  (输出目录)
